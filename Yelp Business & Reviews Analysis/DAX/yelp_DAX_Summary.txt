    RETURN

        BizCount >= 1000000, FORMAT ( DIVIDE ( BizCount, 1000000 ), "0.#" ) & "M",
        BizCount >= 1000, FORMAT ( DIVIDE ( BizCount, 1000 ), "0.#" ) & "K",
        FORMAT ( BizCount, "#,0" )
    )
